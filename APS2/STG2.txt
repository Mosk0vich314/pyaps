        % Methods to use the Camera object

        % Button pushed function: CaptureImageButton
        function captureImage(app, file, path)

            if isrunning(app.Vid)
                % Capture the current frame
                frame = getsnapshot(app.Vid);

                if isequal(file, 0) || isequal(path, 0)
                    uialert(app.ESCOTestUIFigure, 'Image save cancelled.', 'Cancelled');
                else
                    % Full path for the saved image
                    % file = sprintf("%s.png", file);
                    fullFileName = fullfile(path, file);

                    % Save the image
                    imwrite(frame, fullFileName);

                    
                end

            else
                % Notify the user that the camera is not running
                uialert(app.ESCOTestUIFigure, 'Camera is not running.', 'Error');
            end
        end


        function timerCameraDAQ(app, ~, ~)
            app.Frame = peekdata(app.Vid, 1);  % Get latest Frame
            flushdata(app.Vid);  % Flush everything else to not fill up memory

            % Update UI video frame
            if ishandle(app.UIvideo)
                image(0, 0, app.Frame, 'Parent', app.UIvideo, 'HitTest', 'off');  % Put Frame into display
            end

            % Calculate Crosshair Positions
            app.CalcCC();

            % Draw Crosshair if needed
            if app.ShowCrosshairCheckBox.Value && ishandle(app.UIvideo)
                hold(app.UIvideo, "on");
                line(app.UIvideo, [app.cameraSettings.Video.CHC.Xc app.cameraSettings.Video.CHC.Xc], [app.cameraSettings.Video.CHC.Yt app.cameraSettings.Video.CHC.Yb], 'Color', 'red');  % Draw Vertical Line
                line(app.UIvideo, [app.cameraSettings.Video.CHC.Xl app.cameraSettings.Video.CHC.Xr], [app.cameraSettings.Video.CHC.Yc app.cameraSettings.Video.CHC.Yc], 'Color', 'red');  % Draw Horizontal Line
                hold(app.UIvideo, "off");
            end

            % Draw Circle if needed
            if app.ShowCircleCheckBox.Value && ishandle(app.UIvideo)
                rectangle('Parent', app.UIvideo, ...
                    'Position', [app.cameraSettings.Video.CHC.Xc - 20 app.cameraSettings.Video.CHC.Yc - 20 40 40], ...
                    'EdgeColor', 'red', 'Curvature', [1 1]);  % Draw bounding Rectangle
                hold(app.UIvideo, "off");
            end

            % Clear frame to release memory
            clear app.Frame;
        end


        function CalcCC(app)    % Calculate Crosshair Coordinates
            app.cameraSettings.Video.CHC.Xc = app.cameraSettings.Video.VidResX/2 + app.OffsetXSpinner.Value;                       % X-Center
            app.cameraSettings.Video.CHC.Yc = app.cameraSettings.Video.VidResY/2 + app.OffsetYSpinner.Value;                       % Y-Center
            Xleft = app.cameraSettings.Video.CHC.Xc - app.SizeSlider.Value;
            if Xleft < 0
                app.cameraSettings.Video.CHC.Xl = 0;
            else
                app.cameraSettings.Video.CHC.Xl = Xleft;
            end
            Xright = app.cameraSettings.Video.CHC.Xc + app.SizeSlider.Value;
            if Xright > app.cameraSettings.Video.VidResX
                app.cameraSettings.Video.CHC.Xr = app.cameraSettings.Video.VidResX;
            else
                app.cameraSettings.Video.CHC.Xr = Xright;
            end
            Ytop = app.cameraSettings.Video.CHC.Yc - app.SizeSlider.Value;
            if Ytop < 0
                app.cameraSettings.Video.CHC.Yt = 0;
            else
                app.cameraSettings.Video.CHC.Yt = Ytop;
            end
            Ybottom = app.cameraSettings.Video.CHC.Yc + app.SizeSlider.Value;
            if Ybottom > app.cameraSettings.Video.VidResY
                app.cameraSettings.Video.CHC.Yb = app.cameraSettings.Video.VidResY;
            else
                app.cameraSettings.Video.CHC.Yb = Ybottom;
            end
        end

        % =================================================================
        % Generate a structure with device coordinates based on the current
        % selected layout, or a .xlsx file iput by the user for more
        % particular devices

        function app = getCustomCoordinatesFromFile(app)

            % Prompt the user to select an Excel file
            [file, path] = uigetfile({'*.xlsx', 'Excel files (*.xlsx)'}, 'Select the device coordinates file');

            % Check if the user canceled the file selection
            if isequal(file, 0) || isequal(path, 0)
                disp('User canceled the file selection.');
                return;
            else
               filePath = fullfile(path, file); % Concatenate path and file name
            end

            % Read the device coordinates from an Excel file
            dataTable = readtable(filePath);

            % Filter rows where either deviceX or deviceY is missing
            validRows = ~isnan(dataTable.deviceX) & ~isnan(dataTable.deviceY);
            filteredTable = dataTable(validRows, :);

            % Initialize the structure with arrays from the filtered table
            app.deviceCoordinates = struct('deviceID', [], 'deviceX', [], 'deviceY', []);

            app.deviceCoordinates.deviceID = filteredTable.deviceID;
            app.deviceCoordinates.deviceX = filteredTable.deviceX;
            app.deviceCoordinates.deviceY = filteredTable.deviceY;
        end

        % Preallocate arrays in `generateDeviceCoordinates`
        function app = generateDeviceCoordinates(app)
            layout = app.Layout;
            currLayout = app.currentLayout;

            % Preallocate the structure with arrays of appropriate size
            numDevices = layout{currLayout}.getxCells() * layout{currLayout}.getyCells() * layout{currLayout}.getRows() * layout{currLayout}.getColumns();
            app.deviceCoordinates.deviceID = zeros(1, numDevices);
            app.deviceCoordinates.deviceX = zeros(1, numDevices);
            app.deviceCoordinates.deviceY = zeros(1, numDevices);

            deviceID = 1;

            xCellWidth = layout{currLayout}.getColumns() * layout{currLayout}.getWidth() + layout{currLayout}.getxLine();
            yCellHeight = layout{currLayout}.getRows() * layout{currLayout}.getHeight() + layout{currLayout}.getyLine();

            for xcell = 1:layout{currLayout}.getxCells()
                for ycell = 1:layout{currLayout}.getyCells()
                    cellXOrigin = (xcell - 1) * xCellWidth;
                    cellYOrigin = (ycell - 1) * yCellHeight;

                    for column = 1:layout{currLayout}.getColumns()
                        for row = 1:layout{currLayout}.getRows()

                            deviceX = cellXOrigin + (column - 1) * layout{currLayout}.getWidth();
                            deviceY = cellYOrigin + (row - 1) * layout{currLayout}.getHeight();

                            % Assign each new value to the respective field in the structure
                            app.deviceCoordinates.deviceID(deviceID) = deviceID;
                            app.deviceCoordinates.deviceX(deviceID) = deviceX;
                            app.deviceCoordinates.deviceY(deviceID) = deviceY;

                            deviceID = deviceID + 1;
                        end
                    end
                end
            end

            app.deviceCoordinates.diffX = diff(app.deviceCoordinates.deviceX); 
            app.deviceCoordinates.diffY = diff(app.deviceCoordinates.deviceY);

        end


        % =================================================================
        % Needle contact routine

        function contactFlag = contactFunction(app, varargin)

            needleSide = varargin{1};
            if nargin == 3
                axes = varargin{2};
            else
                axes = [];
            end

            contactFlag = 0;

            if ~isempty(axes)
                hold(app.NeedlePlot, "off");
                cla(axes);
                hold(axes, "on");
                line = animatedline('Color','black','LineWidth',1.5,'Parent', axes);
                yline(axes, app.needleContact.threshold, '--r', 'LineWidth', 1.5);
            end

            app.needleContact.startRoutine;

            % plot realtime
            previous_counter = 0;
            pause(0.2);
            run = true; 

            while run && Get_Par(19) > 0 && ~contactFlag

                while (app.flagPause)
                    pause(0.1)
                end
                if (app.flagStop)
                    break;
                end

                run = Process_Status(2);
                actual_time = Get_Par(19) - 1;

                % get current and update plot
                % temp = zeros(1, actual_time - previous_counter);

                try
                    pause(0.1)
                    currentValue = abs(GetData_Double(2, previous_counter + 1, actual_time - previous_counter));
                    if ~isempty(axes)
                        addpoints(line, app.needleContact.Gate.time(previous_counter + 1 : actual_time), currentValue);
                        drawnow limitrate
                    end
                end

                previous_counter = actual_time;

                if sum(currentValue > app.needleContact.threshold) > 0 

                    contactFlag = 1;

                    switch needleSide

                        case 'Left'
                            app.LLamp.Color = 'green';

                        case 'Right'
                            app.RLamp.Color = 'green';
                    end

                else

                    switch needleSide

                        case 'Left'
                            app.Motor.move_leftProbe(-10);

                        case 'Right'
                            app.Motor.move_rightProbe(-10);
                    end

                end

            end

            for i=1:10
                Stop_Process(i);
            end

        end

        function moveDevice(app, devID)
            
            xMove = app.deviceCoordinates.diffX(devID - 1);
            yMove = -app.deviceCoordinates.diffY(devID - 1);

            if abs(xMove) > app.maxStep
                for i = 1:floor(abs(xMove)/app.maxStep)
                    app.Motor.move_x(sign(xMove) * app.maxStep);
                    pause(1);
                end
                app.Motor.move_x(sign(xMove) * mod(abs(xMove), app.maxStep));             
            else
                app.Motor.move_x(xMove);
            end

            if abs(yMove) > app.maxStep
                for i = 1:floor(abs(yMove)/app.maxStep)
                    app.Motor.move_y(sign(yMove) * app.maxStep);
                    pause(1);
                end
                app.Motor.move_y(sign(yMove) * mod(abs(yMove), app.maxStep));
            else

                app.Motor.move_y(yMove);
            end

        end

        function measureChip(app)
            warnString = sprintf('Align on Device %0d', app.DeviceNrEditField.Value);

            mydlg = warndlg('Put Switchbox on Yellow!!', 'Warning');
            waitfor(mydlg);

            % app.Motor.move_leftProbe(50); % Move needles back up before moving
            app.LLamp.Color = 'red';
            % app.Motor.move_rightProbe(50);
            app.RLamp.Color = 'red';
            
            % app.CirclesSpinner.Value = app.Layout{app.currentLayout}.getColumns;
            % app.CirclesSpinner.Value = 3;

            if app.DeviceNrEditField.Value >= (((app.Layout{app.currentLayout}.getxCells - 1) * app.Layout{app.currentLayout}.getyCells * (app.Layout{app.currentLayout}.getRows * app.Layout{app.currentLayout}.getColumns)) + 1)
                isLastColumn = 1;
            else
                isLastColumn = 0;
            end

            if app.isRef_needed
                mydlg = warndlg('Align on top-left device for reference capture', 'Warning');
                waitfor(mydlg);
                app.captureMultiReference(isLastColumn);
            end

            mydlg = warndlg(warnString, 'Warning');
            waitfor(mydlg);

            % Open log file once
            logFilePath = fullfile(app.ADwinSettings.save_dir, 'contact_failures_log.txt');
            logFile = fopen(logFilePath, 'a');  % 'a' mode to append data

            % for devID = app.DeviceNrEditField.Value:(length(app.deviceCoordinates.deviceID) - 1)
            for devID = app.DeviceNrEditField.Value:(length(app.deviceCoordinates.deviceID))
                if app.flagStop
                    break
                end
                while (app.flagPause)
                    pause(0.1);
                end
                if ~(devID == app.DeviceNrEditField.Value)
                    app.moveDevice(devID);
                end
                if mod(devID, (app.Layout{app.currentLayout}.getRows)) == 1 && ~(devID == app.DeviceNrEditField.Value)
                    if app.currentLayout == 1

                        app.moveDevice(devID - 1); %add if very inconsistent between cells

                    elseif app.currentLayout == 2 

                        app.Motor.move_y(-200);

                    end

                    if mod(devID, (app.Layout{app.currentLayout}.getRows * app.Layout{app.currentLayout}.getColumns)) == 1
                        app.runImageCorrection(app.Frame_reference_centers{1});
                        if devID == (((app.Layout{app.currentLayout}.getxCells - 1) * app.Layout{app.currentLayout}.getyCells * (app.Layout{app.currentLayout}.getRows * app.Layout{app.currentLayout}.getColumns)) + 1)
                            isLastColumn = 1;
                            app.captureMultiReference(isLastColumn);
                            app.Motor.move_x((-app.Layout{app.currentLayout}.getColumns) * app.Layout{app.currentLayout}.getWidth);
                        end
                    elseif mod(devID, (app.Layout{app.currentLayout}.getRows * app.Layout{app.currentLayout}.getColumns)) == 6
                        if isLastColumn
                            app.CirclesSpinner.Value = app.CirclesSpinner.Value - 1;
                            app.runImageCorrection(app.Frame_reference_centers{2});
                            app.CirclesSpinner.Value = app.CirclesSpinner.Value + 1;
                        else
                            app.runImageCorrection(app.Frame_reference_centers{2});
                        end
                    elseif mod(devID, (app.Layout{app.currentLayout}.getRows * app.Layout{app.currentLayout}.getColumns)) == 11
                        if isLastColumn
                            app.CirclesSpinner.Value = app.CirclesSpinner.Value - 2;
                            app.runImageCorrection(app.Frame_reference_centers{3});
                            app.CirclesSpinner.Value = app.CirclesSpinner.Value + 2;
                        else
                            app.runImageCorrection(app.Frame_reference_centers{3});
                        end
                    end
                end

                if app.flagStop

                    clear all;
                    close all;
                    clc;
                   
                    break
                    
                end

                pause(2);

                while (app.flagPause)
                    pause(0.1);
                end

                if ~app.testingMovement
                    contactFlag = app.contactFunction('Left');
                    if ~contactFlag
                        app.Motor.move_leftProbe(30);
                        contactFlag = app.contactFunction('Left');
                        if ~contactFlag
                            fprintf(logFile, 'Left contact failed on device %d\n', devID);  % Log the failure
                            app.failedDevices = app.failedDevices + 1;
                            app.Motor.move_leftProbe(50);
                            app.DeviceNrEditField.Value = devID;
                            continue;
                        end
                    end
                    app.switchingBox.startRoutine;
                    pause(2);
                    contactFlag = app.contactFunction('Right');
                    if ~contactFlag
                        app.Motor.move_rightProbe(30);
                        contactFlag = app.contactFunction('Right');
                        if ~contactFlag
                            fprintf(logFile, 'Right contact failed on device %d\n', devID);  % Log the failure
                            app.failedDevices = app.failedDevices + 1;
                            app.Motor.move_rightProbe(50);
                            app.switchingBox.startRoutine;
                            app.DeviceNrEditField.Value = devID;
                            continue;
                        end
                    end
                    app.switchingBox.startRoutine;
                    pause(2);
                    app.executeRoutines(app.ADwinSettings.save_dir, devID);

                    app.Motor.move_leftProbe(30); % Move needles back up before moving
                    app.LLamp.Color = 'red';
                    app.Motor.move_rightProbe(30);
                    app.RLamp.Color = 'red';
                end

                app.DeviceNrEditField.Value = devID;
            end

            % Close log file
            fclose(logFile);

            if ~app.flagStop
                warndlg('Chip done!', 'Warning');
            else
                warndlg('Scan aborted!', 'Warning');
            end
        end

        function [Centers, Radii, Image_data_red] = detectCircles(app)

            close all

            % get frame
            frame = getsnapshot(app.Vid);
            resizedFrame = imresize(frame, 0.25);

            % Cut out the rightmost 10% of the resized frame
            [~, cols, ~] = size(resizedFrame);
            if app.currentLayout == 1

                cutCols = round(cols * 0.9);

            elseif app.currentLayout == 2

                cutCols = round(cols * 1);

            end

            resizedFrame = resizedFrame(:, 1:cutCols, :);

            Image_data_red = adapthisteq(resizedFrame(:,:,1));


            pixelTolerance = 10;

            % loop over sensitivity to find number of circles

            Radii = [];
            Sensitivity = app.SensitivitySpinner.Value;
            correctCircles = 0;
            while Sensitivity < 1 && ~correctCircles

                [Centers, Radii] = imfindcircles(Image_data_red,[app.MinRadiuspxSpinner.Value app.MaxRadiuspxSpinner.Value],'ObjectPolarity','bright','Sensitivity',Sensitivity);              

                if numel(Radii) >= app.CirclesSpinner.Value

                    % Initialize an empty matrix to hold the groups of valid circles
                    validGroups = [];

                    % Number of circles detected
                    N = size(Centers, 1);
                    numCircles = app.CirclesSpinner.Value;

                    if numCircles == 1
                        % If only one circle is needed, all detected circles are valid
                        validGroups = (1:N)';
                    else
                        % Check each combination of circles
                        combs = nchoosek(1:N, numCircles);
                        for c = 1:size(combs, 1)
                            groupIndices = combs(c, :);
                            yCoords = Centers(groupIndices, 2);

                            % Check if these centers are approximately on the same y-line
                            if all(abs(yCoords - mean(yCoords)) <= pixelTolerance)
                                % Store the indices of the valid group
                                validGroups = [validGroups; groupIndices];
                            end
                        end
                    end

                    % If any valid groups are found, select the closest group of circles
                    if ~isempty(validGroups)
                        if size(validGroups, 1) == 1
                            bestGroup = validGroups;
                        else
                            minMaxDistance = inf;
                            bestGroup = [];

                            for g = 1:size(validGroups, 1)
                                groupIndices = validGroups(g, :);
                                xCoords = Centers(groupIndices, 1);
                                maxDistance = max(xCoords) - min(xCoords);

                                if maxDistance < minMaxDistance
                                    minMaxDistance = maxDistance;
                                    bestGroup = groupIndices;
                                end
                            end
                        end

                        % Filter the centers to only include those in the best group
                        Centers = Centers(bestGroup, :);
                        correctCircles = 1;
                    else
                        correctCircles = 0;
                    end
                end

                Sensitivity = Sensitivity + 0.005;
            end

            fprintf('Found %02d circles', numel(Radii));
            Radii

            % sort by x position
            [~, idx] = sort(Centers(:,1));
            Centers = [Centers(idx, 1) Centers(idx, 2)];
            Radii = Radii(idx);

        end

        function [ref_centers, ref_radii] = captureReference(app)

            [Centers, Radii, Image_data_red] = app.detectCircles;
        
            ref_centers = Centers;
            ref_radii = Radii;

            % show output

                end
            end

        end

                    end

            else

                saver = app.CirclesSpinner.Value;

                for i = 1:(app.Layout{app.currentLayout}.getColumns)

                    if ~app.flagStop

                        [app.Frame_reference_centers{i}, app.Frame_reference_radii{i}] = app.captureReference;
                        app.Motor.move_x(app.Layout{app.currentLayout}.getWidth);

                    end

                    app.CirclesSpinner.Value = app.CirclesSpinner.Value - 1;

                end

                app.CirclesSpinner.Value = saver;

            end

        end

        function runImageCorrection(app, ref_centers)

            counter = 1;

            error = 100;
            while max(abs(error)) > 5 && counter < 10

                [Centers, Radii, ~] = app.detectCircles;

                % close all

                % get error
                try
                    if(numel(Centers(:, 1)) > 1)
                        error = mean(ref_centers - Centers);
                    else
                        error = ref_centers - Centers;
                    end

                    % convert error to um
                    error_um = error / mean(Radii) * app.CirclesizeSpinner.Value;

                    % move stage
                    if app.flagStop
                        break
                    end
                    app.Motor.move_x(-error_um(1));
                    app.Motor.move_y(error_um(2));

                    % update counter
                    counter = counter + 1;
                end



                % dlg = warndlg('Move?');
                % waitfor(dlg);
            end

        end

        % Button pushed function: Button_Align_rotation
        function Button_Align_rotationPushed(app, event)

            app.flagStop = 0;
            
            groupMove_x = app.Layout{app.currentLayout}.getColumns * app.Layout{app.currentLayout}.getWidth + app.Layout{app.currentLayout}.getxLine;
            range = app.RangeSpinner.Value;
            step = app.StepSpinner.Value;

            rotRange = -range:step:range;
            rotSteps = [-range step*ones(1, numel(rotRange) - 1)];
            error_y = inf(numel(rotSteps), 1);  % Initialize with large values
            % Capture initial position to readjust every time and improve
            % stability
            app.Button_CaptureReferencePushed;

            for counter = 1:numel(rotSteps)
                % Move the stage by the current rotation step
                app.Motor.move_theta(rotSteps(counter));

                % Capture this reference
                [ref_centers, ~] = app.detectCircles;

                if app.flagStop
                    break
                end

                app.Motor.move_x(groupMove_x);

                [Centers, Radii, ~] = app.detectCircles;
                
                % Calculate the error
                error = mean(ref_centers - Centers);

                % Convert error to micrometers
                error_um = error / mean(Radii) * app.CirclesizeSpinner.Value;
                error_y(counter) = abs(error_um(2));

                % Log the error for debugging
                fprintf('Rotation step %d: error = %f um\n', rotRange(counter), error_y(counter));

                app.Motor.move_x(-groupMove_x);
                app.Button_Run_correctionPushed;
            end

            if ~app.flagStop
                % Find the rotation step with the minimum error
                [~, minIndex] = min(error_y);
                % Move the stage to the optimal rotation angle
                app.Motor.move_theta((numel(rotSteps) - minIndex)*(-step));

                warndlg('Rotation adjusted!', 'Warning');
            else
                warndlg('Rotation not adjusted!', 'Warning');
            end




        end

        % Button pushed function: GenerateSpreadsheetButton
        function GenerateSpreadsheetButtonPushed(app, event)

            % Function to process yield data
            function devArray = processYield(folderPath, sampleName)
                % Get a list of all .mat files in the specified folder
                filePattern = fullfile(folderPath, ['*' sampleName '-*.mat']);
                files = dir(filePattern);
                numFiles = length(files);

                % Initialize the gapOpening array
                devArray = ones(1, app.CellcolumnsEditField.Value * app.CellrowsEditField.Value * app.Layout{app.currentLayout}.getRows() * app.Layout{app.currentLayout}.getColumns());
                selected_voltage = 0.5;
                % Loop over each file
                for k = 1:numFiles
                    fileName = files(k).name;

                    % Extract the sample number from the file name
                    tokens = regexp(fileName, [sampleName '-(\d+)_IV'], 'tokens');
                    % Check if filename is correctly formatted
                    if isempty(tokens)
                        continue; % Skip this iteration if tokens is empty
                    end
                    sampleNumber = str2double(tokens{1}{1});

                    % Load the .mat file
                    data = load(fullfile(folderPath, fileName));

                    % Access the 'IV' class and the 'current' cell array
                    if isfield(data, 'IV') && isfield(data.IV, 'current')
                        currentCellArray = data.IV.current;

                        % Perform Desired Calculation
                        computedValues = ones(size(currentCellArray));
                        for j = 1:numel(currentCellArray)

                            % Load the data
                            bias_voltage = data.IV.bias;
                            current = currentCellArray{j};

                            switch app.spreadsheetDropDown.Value

                                case 'Gap Opening'

                                    if mean(abs(currentCellArray{j})) > 1e-11
                                        computedValues(j) = 0;
                                    end

                                case 'Linearized - Resistance'

                                    % Calculate the derivative of the current with respect to bias voltage
                                    dI_dV = diff(current) ./ diff(bias_voltage);

                                    % Filter out infinite and NaN values in the derivative
                                    valid_indices = isfinite(dI_dV);
                                    filtered_dI_dV = dI_dV(valid_indices);

                                    % Smooth the derivative to reduce noise
                                    avg_dI_dV = movmean(filtered_dI_dV, 5);

                                    % Identify the region where the derivative is approximately constant
                                    threshold = std(avg_dI_dV) * 0.1;
                                    linear_region = abs(avg_dI_dV - mean(avg_dI_dV)) < threshold;

                                    linear_bias_voltage = bias_voltage(linear_region);
                                    linear_current = current(linear_region);

                                    % Perform linear regression on the linear region
                                    p = polyfit(linear_bias_voltage, linear_current, 1);
                                    slope = p(1);

                                    % Calculate the resistance
                                    computedValues(j) = abs( 1 / slope );

                                case 'Resistance - 04V'
                                    
                                    % Define the linearization region 
                                    linearization_setpoint = 0.4;
                                    linearization_threshold = 0.05;

                                    % Select the data points in the linearization region
                                    linearization_indices = (linearization_setpoint - linearization_threshold < bias_voltage) & (bias_voltage < linearization_setpoint + linearization_threshold);

                                    linearization_voltage = bias_voltage(linearization_indices);
                                    linearization_current = current(linearization_indices);

                                    p = polyfit(linearization_voltage, linearization_current, 1); % p(1) is the slope, p(2) is the intercept

                                    % Calculate resistance as the inverse of the slope
                                    computedValues(j) =abs( 1 / p(1) );

                                case 'Low Bias - Resistance'

                                    % Define the low-bias region threshold
                                    low_bias_threshold = 0.1; % Adjust this value based on your data

                                    % Select the data points in the low-bias region
                                    low_bias_indices = abs(bias_voltage) < low_bias_threshold;
                                    low_bias_voltage = bias_voltage(low_bias_indices);
                                    low_bias_current = current(low_bias_indices);

                                    % Perform linear regression on the low-bias region
                                    p = polyfit(low_bias_voltage, low_bias_current, 1); % p(1) is the slope, p(2) is the intercept

                                    % Calculate resistance as the inverse of the slope
                                    computedValues(j) =abs( 1 / p(1) );

                                case 'Linearity'

                                    % Fit a linear regression model
                                    p = polyfit(bias_voltage, current, 1);
                                    lin_current = polyval(p, bias_voltage);

                                    % Calculate residuals
                                    residuals = current - lin_current;

                                    % Calculate the sum of squared residuals
                                    ss_res = sum(residuals.^2);

                                    % Calculate the total sum of squares (total variability in the data)
                                    ss_tot = sum((current - mean(current)).^2);

                                    % Calculate R-squared
                                    R_squared = 1 - (ss_res / ss_tot);

                                    computedValues(j) = R_squared;

                                case 'Current'                                    
                              
                                    % Find indices of the specific voltage in the voltage vector
                                    indices = find(bias_voltage == selected_voltage);

                                    if ~isempty(indices)
                                        % If the specific voltage is found, calculate the average current
                                        computedValues(j) = mean(current(indices));
                                    else
                                        % If the specific voltage is not found, find the closest voltage
                                        [~, closestIndex] = min(abs(bias_voltage - selected_voltage));
                                        computedValues = current(closestIndex);
                                    end

                            end % switch

                        end % for j

                        % Store results
                        devArray(sampleNumber) = computedValues;
                    else
                        warning('IV class or current array not found in %s', fileName);
                    end % if
                end % for k
            end % function


            % Helper function to convert column numbers to Excel column letters
            function letter = columnLetter(columnNumber)
                letter = '';
                while columnNumber > 0
                    remainder = mod(columnNumber - 1, 26);
                    letter = [char(65 + remainder) letter];
                    columnNumber = floor((columnNumber - remainder - 1) / 26);
                end
            end

            % Define the necessary variables (Adjust these as needed)
            folderPath = app.ADwinSettings.save_dir; % Replace with the correct folder path
            sampleName = app.ADwinSettings.sample; % Replace with the correct sample name

            % % Load the gapOpening data
            devArray = processYield(folderPath, sampleName);

            % Initialize the 3D cell array based on the layout
            Cellcolumns = app.CellcolumnsEditField.Value;
            Cellrows = app.CellrowsEditField.Value;
            numRows = app.Layout{app.currentLayout}.getRows();
            numCols = app.Layout{app.currentLayout}.getColumns();

            devMatrix3D = ones(Cellrows * numRows, Cellcolumns * numCols);
            devNumbers3D = zeros(Cellrows * numRows, Cellcolumns * numCols);

            flag = 0;
            n = 1;

            for kY = 1:Cellcolumns
                for kX = 1:Cellrows
                    if flag
                        break;
                    end
                    for colIndex = 1:numCols
                        if flag
                            break;
                        end
                        for rowIndex = 1:numRows
                            if n <= numel(devArray)
                                devMatrix3D((kX - 1) * numRows + rowIndex, (kY - 1) * numCols + colIndex) = devArray(n);
                                devNumbers3D((kX - 1) * numRows + rowIndex, (kY - 1) * numCols + colIndex) = n;
                                n = n + 1;
                            else
                                flag = 1;
                                break;
                            end
                        end
                    end
                end
            end

            % Save the 3D cell array
            matname = sprintf('%s-%s.mat', app.ADwinSettings.sample, app.spreadsheetDropDown.Value);
            matname = fullfile(folderPath, matname);
            save(matname, "devMatrix3D");

            % Write to Excel file
            filename = sprintf('%s-%s.xlsx', app.ADwinSettings.sample, app.spreadsheetDropDown.Value);
            filename = fullfile(folderPath, filename);


            % Initialize an empty cell array to store the logical values for the entire layout
            fullLayout = devMatrix3D'; % Note the swapped dimensions            

            switch app.spreadsheetDropDown.Value

                case 'Gap Opening'

                    try
                       
                        excelData = devNumbers3D';
                        writematrix(excelData, filename);
                        excel = actxserver('Excel.Application');
                        workbook = excel.Workbooks.Open(filename);
                        sheet = workbook.Sheets.Item(1);

                        % Define the color codes
                        greenColor = hex2dec('00FF00'); % Hex for green
                        redColor = hex2dec('0000FF');   % Hex for red

                        % Apply formatting to each cell
                        for row = 1:size(excelData, 1)

                            for col = 1:size(excelData, 2)

                                cellAddress = sprintf('%s%d', columnLetter(col), row);
                                cell = sheet.Range(cellAddress);

                                if fullLayout(row, col) == true

                                    cell.Interior.Color = greenColor;

                                else

                                    cell.Interior.Color = redColor;

                                end

                            end

                        end

                        % % Force a screen update
                        % excel.ScreenUpdating = true;
                        % excel.DisplayAlerts = false;

                        % Save and close the workbook
                        workbook.Save();
                        workbook.Close(false);
                        excel.Quit();

                        % Clean up
                        delete(excel);

                    catch ME
                        % Display any errors that occur
                        disp('Error applying conditional formatting to Excel cells:');
                        disp(ME.message);
                        % Ensure Excel process is quit in case of error
                        if exist('excel', 'var')

                           excel.Quit();
                            delete(excel);

                        end

                    end

                otherwise

                    excelData = fullLayout;
                    writematrix(excelData, filename);

            end

            warndlg('Spreadsheet generated!', 'Success!');
          
        end

                % app.Motor.corrFactorX = 1;
                % app.Motor.corrFactorY = 1;

                app.Button_CaptureReferencePushed;
                % =================================================================
                % Adjust stepFactorX
                % =================================================================
                steps = app.Layout{app.currentLayout}.getColumns * app.Layout{app.currentLayout}.getWidth + app.Layout{app.currentLayout}.getxLine;

                if app.flagStop
                    app.Motor.stepFactorX = app.StepfactorXdirectionEditField.Value;
                    app.Motor.stepFactorY = app.StepfactorYdirectionEditField.Value;
                    app.Motor.corrFactorX = app.HysthfactorXdirectionEditField.Value;
                    app.Motor.corrFactorY = app.HysthfactorYdirectionEditField.Value;

                    break
                end

                app.Motor.move_x(steps);
                
                [Centers, Radii, ~] = app.detectCircles;

                if app.flagStop
                    app.Motor.stepFactorX = app.StepfactorXdirectionEditField.Value;
                    app.Motor.stepFactorY = app.StepfactorYdirectionEditField.Value;
                    app.Motor.corrFactorX = app.HysthfactorXdirectionEditField.Value;
                    app.Motor.corrFactorY = app.HysthfactorYdirectionEditField.Value;

                    break
                end

                % get error
                try

                    error = mean(app.tmp_ref_centers - Centers);

                    % convert error to um
                    error_um = error / mean(Radii) * app.CirclesizeSpinner.Value;
                    deltaSteps = error_um(1);
                    correction = (steps - deltaSteps)/steps;

                    app.Motor.stepFactorX = app.Motor.stepFactorX*correction;
                catch
                    mydlg = warndlg('stepFactorX not updated', 'Warning');
                    waitfor(mydlg);

                end

                app.StepfactorXdirectionEditField.Value = app.Motor.stepFactorX;
                app.Button_Run_correctionPushed;
                % =================================================================
                % Adjust corrFactorX
                % =================================================================              
                steps = -steps;
                if app.flagStop
                    app.Motor.stepFactorX = app.StepfactorXdirectionEditField.Value;
                    app.Motor.stepFactorY = app.StepfactorYdirectionEditField.Value;
                    app.Motor.corrFactorX = app.HysthfactorXdirectionEditField.Value;
                    app.Motor.corrFactorY = app.HysthfactorYdirectionEditField.Value;

                    break
                end
                app.Motor.move_x(steps);
                
                [Centers, Radii, ~] = app.detectCircles;

                if app.flagStop
                    app.Motor.stepFactorX = app.StepfactorXdirectionEditField.Value;
                    app.Motor.stepFactorY = app.StepfactorYdirectionEditField.Value;
                    app.Motor.corrFactorX = app.HysthfactorXdirectionEditField.Value;
                    app.Motor.corrFactorY = app.HysthfactorYdirectionEditField.Value;

                    break
                end

                % get error
                try

                    error = mean(app.tmp_ref_centers - Centers);

                    % convert error to um
                    error_um = error / mean(Radii) * app.CirclesizeSpinner.Value;
                    deltaSteps = error_um(1);
                    correction = (steps - deltaSteps)/steps;
                    app.Motor.corrFactorX = app.Motor.corrFactorX*correction;

                catch
                    mydlg = warndlg('corrFactorX not updated', 'Warning');
                    waitfor(mydlg);

                end

                app.HysthfactorXdirectionEditField.Value = app.Motor.corrFactorX;
                app.Button_Run_correctionPushed;
                % =================================================================
                % Adjust stepFactorY
                % =================================================================
                steps = (-1)*(app.Layout{app.currentLayout}.getRows * app.Layout{app.currentLayout}.getHeight + app.Layout{app.currentLayout}.getyLine);
                if app.currentLayout == 1
                    extraStep = (-1) * app.Layout{app.currentLayout}.getHeight;
                elseif app.currentLayout == 2
                    extraStep = (-1) * 200;
                end

                if app.flagStop
                    app.Motor.stepFactorX = app.StepfactorXdirectionEditField.Value;
                    app.Motor.stepFactorY = app.StepfactorYdirectionEditField.Value;
                    app.Motor.corrFactorX = app.HysthfactorXdirectionEditField.Value;
                    app.Motor.corrFactorY = app.HysthfactorYdirectionEditField.Value;

                    break
                end
                
                app.Motor.move_y(extraStep);

                app.Button_CaptureReferencePushed;

                app.Motor.move_y(steps);              
                               
                [Centers, Radii, ~] = app.detectCircles;

                if app.flagStop
                    app.Motor.stepFactorX = app.StepfactorXdirectionEditField.Value;
                    app.Motor.stepFactorY = app.StepfactorYdirectionEditField.Value;
                    app.Motor.corrFactorX = app.HysthfactorXdirectionEditField.Value;
                    app.Motor.corrFactorY = app.HysthfactorYdirectionEditField.Value;

                    break
                end

                % get error
                try

                    error = mean(app.tmp_ref_centers - Centers);

                    % convert error to um
                    error_um = error / mean(Radii) * app.CirclesizeSpinner.Value;
                    deltaSteps = error_um(2);
                    correction = (steps + deltaSteps)/(steps);

                    app.Motor.stepFactorY = app.Motor.stepFactorY*correction;
                catch
                    mydlg = warndlg('stepFactorY not updated', 'Warning');
                    waitfor(mydlg);

                end

                app.StepfactorYdirectionEditField.Value = app.Motor.stepFactorY;
                app.Button_Run_correctionPushed;
                % =================================================================
                % Adjust corrFactorY
                % =================================================================
                steps = -steps;

                if app.flagStop
                    app.Motor.stepFactorX = app.StepfactorXdirectionEditField.Value;
                    app.Motor.stepFactorY = app.StepfactorYdirectionEditField.Value;
                    app.Motor.corrFactorX = app.HysthfactorXdirectionEditField.Value;
                    app.Motor.corrFactorY = app.HysthfactorYdirectionEditField.Value;

                    break
                end

                app.Motor.move_y(steps);
                
                [Centers, Radii, ~] = app.detectCircles;

                if app.flagStop
                    app.Motor.stepFactorX = app.StepfactorXdirectionEditField.Value;
                    app.Motor.stepFactorY = app.StepfactorYdirectionEditField.Value;
                    app.Motor.corrFactorX = app.HysthfactorXdirectionEditField.Value;
                    app.Motor.corrFactorY = app.HysthfactorYdirectionEditField.Value;

                    break
                end

                % get error
                try

                    error = mean(app.tmp_ref_centers - Centers);

                    % convert error to um
                    error_um = error / mean(Radii) * app.CirclesizeSpinner.Value;
                    deltaSteps = error_um(2);
                    correction = (steps + deltaSteps)/(steps);

                    app.Motor.corrFactorY = app.Motor.corrFactorY*correction;

                catch
                    mydlg = warndlg('corrFactorY not updated', 'Warning');
                    waitfor(mydlg);

                end

                app.HysthfactorYdirectionEditField.Value = app.Motor.corrFactorY;
                app.Button_Run_correctionPushed;

                % =================================================================
                % Adjust Rotation
                % =================================================================
                if strcmp(app.RotationSwitch.Value, 'Yes')
                    app.Button_Align_rotationPushed;
                end

                app.flagStop = 1;

                warndlg('Initialization done!', 'Success');

            end

            if ~app.flagStop

                warndlg('Initialization aborted!', 'Warning');
            
            end

        end

        function GenerateHysthogramButtonPushed(app, event)
            % filename = sprintf('%s-%s.xlsx', app.ADwinSettings.sample, app.spreadsheetDropDown.Value);
            % filename = fullfile(app.ADwinSettings.save_dir, filename);
            % % Read the data from the Excel file
            % data = readmatrix(filename);
            % 
            % % Flatten the matrix and remove NaN values
            % values = data(~isnan(data(:)));
            % 
            % % Determine the orders of magnitude in the dataset
            % min_order = floor(log10(min(values)));
            % max_order = ceil(log10(max(values)));
            % 
            % % Calculate the number of bins
            % num_bins = max_order - min_order + 1;
            % if num_bins < 5
            %     % If less than 5 orders of magnitude, split some bins within the same order
            %     bin_edges = linspace(min_order, max_order, 5 + 1);
            %     bins = 10.^bin_edges;
            % else
            %     % Use one bin per order of magnitude
            %     bins = 10 .^ (min_order:max_order);
            % end
            % 
            % % Create the histogram
            % figure;
            % histogram(values, bins);
            % 
            % % Set logarithmic scale for x-axis
            % set(gca, 'XScale', 'log');
            % 
            % % Set axis labels and title
            % xlabel('Resistance (Ohms)');
            % ylabel('Frequency');
            % title('Histogram of Resistance Values');
            % 
            % % Save the histogram as a JPEG file
            % filename = sprintf('%s_%s Hysthogram.jpg', app.ADwinSettings.sample, app.spreadsheetDropDown.Value);
            % filename = fullfile(app.ADwinSettings.save_dir, filename);
            % saveas(gcf,filename)
            % % Close the figure
            % successDlg = warndlg('Hysthogram generated!', 'Success!');
            % waitfor(successDlg);
            % close(gcf);

            filename = sprintf('%s-%s.xlsx', app.ADwinSettings.sample, app.spreadsheetDropDown.Value);
            filename = fullfile(app.ADwinSettings.save_dir, filename);
            % Read the resistance data from the Excel file
            data = readmatrix(filename);

            % Read the color data from the Excel file (using a method to extract colors)
            gapOpeningName = sprintf('%s-Gap Opening.xlsx', app.ADwinSettings.sample);
            gapOpeningName = fullfile(app.ADwinSettings.save_dir, gapOpeningName);
            [~, ~, raw] = xlsread(gapOpeningName);

            % Initialize a logical mask to filter out green cells
            green_mask = false(size(data));

            % Check each cell for green color (this example assumes that green cells
            % have specific RGB values in Excel)
            for i = 1:size(raw, 1)
                for j = 1:size(raw, 2)
                    if iscell(raw{i, j}) && ischar(raw{i, j})
                        % Check if the cell's string includes the color 'green'
                        if contains(lower(raw{i, j}), 'green')
                            green_mask(i, j) = true;
                        end
                    end
                end
            end

            % Flatten the matrix and remove NaN values and green cells
            values = data(~isnan(data(:)) & ~green_mask(:));

            % Determine the orders of magnitude in the dataset
            min_order = floor(log10(min(values)));
            max_order = ceil(log10(max(values)));

            % Calculate the number of bins
            num_bins = max_order - min_order + 1;

            if num_bins < 5
                % If less than 5 orders of magnitude, split some bins within the same order
                bin_edges = linspace(min_order, max_order, 5 + 1);
                bins = 10.^bin_edges;
            else
                % Use one bin per order of magnitude
                bins = 10 .^ (min_order:max_order);
            end

            % Create the histogram
            figure;
            histogram(values, bins);

            % Set logarithmic scale for x-axis
            set(gca, 'XScale', 'log');

            % Set axis labels and title
            switch app.spreadsheetDropDown.Value
                case 'Linearized - Resistance'
                    xlabel('Resistance (Ohms)');
                    ylabel('Frequency');
                    title('Histogram of Linearized Resistance Values');
                case 'Low Bias - Resistance'
                    xlabel('Resistance (Ohms)');
                    ylabel('Frequency');
                    title('Histogram of Low Bias Resistance Values');
                case 'Current'
                    xlabel('Current (Ampere)');
                    ylabel('Frequency');
                    title('Histogram of Current Values');
            end

            % Save the histogram as a JPEG file
            filename = sprintf('%s-%s Hysthogram.jpg', app.ADwinSettings.sample, app.spreadsheetDropDown.Value);
            filename = fullfile(app.ADwinSettings.save_dir, filename);
            % Save the histogram as a JPEG file
            saveas(gcf, filename);

            % Close the figure
            successDlg = warndlg('Hysthogram generated!', 'Success!');
            waitfor(successDlg);
            close(gcf);


        end

        % Value changed function: LabelEditField
        function LabelEditFieldValueChanged(app, event)



            label = app.LabelEditField.Value;

            % Use regular expression to extract the letter and number
            tokens = regexp(label, '([A-Za-z])(\d+)', 'tokens');

            % Assign to separate variables
            letter = tokens{1}{1};           % The letter part
            letter = upper(letter); % Convert to uppercase to handle lowercase letters
            number = str2double(tokens{1}{2});  % The number part as a double

            col = double(letter) - double('A') + 1;
            row = number;

            Cellcolumns = app.CellcolumnsEditField.Value;
            Cellrows = app.CellrowsEditField.Value;
            numRows = app.Layout{app.currentLayout}.getRows();
            numCols = app.Layout{app.currentLayout}.getColumns();

            devNumbers3D = zeros(Cellrows * numRows, Cellcolumns * numCols);
            devNumbers3D = devNumbers3D';
            flag = 0;
            n = 1;

            for kY = 1:Cellcolumns
                for kX = 1:Cellrows
                    if flag
                        break;
                    end
                    for colIndex = 1:numCols
                        if flag
                            break;
                        end
                        for rowIndex = 1:numRows
                            if n <= numel(devNumbers3D)
                                devNumbers3D((kY - 1) * numCols + colIndex, (kX - 1) * numRows + rowIndex) = n;
                                n = n + 1;
                            else

        end

        % Changes arrangement of the app based on UIFigure width
        function updateAppLayout(app, event)
            currentFigureWidth = app.ESCOTestUIFigure.Position(3);
            if(currentFigureWidth <= app.onePanelWidth)
                % Change to a 2x1 grid
                app.GridLayout.RowHeight = {480, 480};
                app.GridLayout.ColumnWidth = {'1x'};
                app.RightPanel.Layout.Row = 2;
                app.RightPanel.Layout.Column = 1;
            else
                % Change to a 1x2 grid
                app.GridLayout.RowHeight = {'1x'};
                app.GridLayout.ColumnWidth = {545, '1x'};
                app.RightPanel.Layout.Row = 1;
                app.RightPanel.Layout.Column = 2;
            end
        end
    end
